void setup() {
  pinMode(13, OUTPUT);
  digitalWrite(13, HIGH);
  system("telnetd -l /bin/sh");
  digitalWrite(13, LOW);
  delay(250);
  digitalWrite(13, HIGH);
  system("ifconfig eth0 169.254.1.1 netmask 255.255.0.0 up");
  digitalWrite(13, LOW);
  delay(250);
  digitalWrite(13, HIGH);
  system("ifconfig wlan0 up");
  digitalWrite(13, LOW);
  delay(250);
  digitalWrite(13, HIGH);
  system("ifup wlan0");
  digitalWrite(13, LOW);
  delay(250);
  digitalWrite(13, HIGH);
  system("cd [DIRECTORY/TO/WEB/SERVER] ; node [SERVER_FILE.js]");
  digitalWrite(13, LOW);
  delay(250);
}
void loop() {
  digitalWrite(13, HIGH);
  delay(1000);
  digitalWrite(13, LOW);
  delay(1000);
}
